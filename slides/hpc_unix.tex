\begin{numberedframe}{\texttt{ls}, \texttt{touch}}

  \item Maybe your account is still empty: do \lstinline{touch newfile},

  \item Options: \lstinline{ls -l} or for specific file \lstinline{ls -l newfile}.
  \end{itemize}

\begin{numberedframe}{Display / add to file: \texttt{cat}}
  \label{sl-lnx:cat}
  \begin{itemize}

  \item Put something in a file: \lstinline{cat > myfile}\\

  \item Now \lstinline{cat} it again.
  \item Do \lstinline{cat >> myfile} and enter some text.

  \end{itemize}

\begin{numberedframe}{\texttt{cp}, \texttt{mv}, \texttt{rm}}
  \label{sl-lnx:cpmv}
  \begin{itemize}
  \item Copy: \lstinline{cp file1 file2}\\
    Do this, check that it's indeed a copy.
  \item Rename or `move': \lstinline{mv file1 file2}\\
    check that the original file doesn't exist any more.
  \item Remove: \lstinline{rm myfile}\\
    This is irrevocable!
  \end{itemize}
\end{numberedframe}

\Level 0 {Directories}

\begin{numberedframe}{Directories}
  \label{sl-lnx:cd}
  \begin{itemize}
  \item Make a subdirectory `folder': \lstinline{mkdir newdir}
  \item Check where you are: \lstinline{pwd}
  \item Now go to the new directory: \lstinline{cd newdir} and \lstinline{pwd}\\
    `change directory' and `present working directory'
  \item Back to your home directory: \lstinline{cd} without further arguments.
  \end{itemize}
\end{numberedframe}

\begin{numberedframe}{Paths}
  \label{sl-lnx:path}
  \begin{itemize}
  \item Do:
    \begin{enumerate}
    \item \lstinline{cd newdir} \item \lstinline{touch nested_file} \item \lstinline{cd}
    \end{enumerate}
  \item Now: \lstinline{ls newdir/nested_file}
  \item That is called a path
    \begin{itemize}
    \item Relative path: does not start with slash
    \item Absolute path (such as \lstinline{pwd} output): starts at root
    \end{itemize}
  \item Create multiple directories:\\
    \lstinline{mkdir -p sub1/sub2/sub3} (\lstinline{p}~for `parent')
  \end{itemize}
\end{numberedframe}

\Level 0 {Redirection, pipes}

\begin{numberedframe}{In/Output redirection}
Output into a file:
\begin{lstlisting}
ls -l > listing
\end{lstlisting}
Append:
\begin{lstlisting}
ls dir1 > dirlisting  
ls dir2 >> dirlisting  
\end{lstlisting}
Input:
\begin{lstlisting}
myprogram < myinput
\end{lstlisting}
\end{numberedframe}

\begin{exerciseframe}{}
  Make a copy of a file, using redirection (so no \lstinline{cp} command).

  Can you use both \verb+>+ and \verb+<+?
\end{exerciseframe}

\begin{numberedframe}{Redirection, formal aspects}
  \label{sl-lnx:redirect}
  \begin{itemize}
  \item There are three standard files: `standard input/output/error'
    (available in~C/C++ as \lstinline{stdin}, \lstinline{stdout}, \lstinline{stderr})
  \item Normally connected to keyboard, screen, and screen respectively.
  \item Redirection: standard out to file:\\
    \lstinline{ls > directorycontents}\\
    (actually, screen is a file, so it is really a \textbf{re}direct)
  \item Standard in from file:
    \lstinline{mail < myfile}\\
    (actually, the keyboard is also a file, so again \textbf{re}direction)
  \end{itemize}
\end{numberedframe}

\begin{numberedframe}{Advanced: splitting out and err}
  \label{sl-lnx:stderr}
  \begin{itemize}
  \item Sometimes you want to split standard out and error:
  \item Use \lstinline{stdout}$=1$ and \lstinline{stderr}$=2$:\\
    \lstinline{myprogram 1>results.out 2>results.err}
  \item Very useful: get rid of errors:\\
    \lstinline{myprogram 2>/dev/null}
  \end{itemize}
\end{numberedframe}

\begin{numberedframe}{Pipes}
  \label{sl-lnx:pipe}
  \begin{itemize}
  \item Redirection is command-to-file.
  \item Pipe: command-to-command\\
    \lstinline{ls | wc -l}\\
    (what does this do?)
  \item Unix philosophy: small building blocks, put together.
  \end{itemize}
\end{numberedframe}

\begin{numberedframe}{More command sequencing}
  \label{sl-lnx:backtick}
  More complicated case of one command providing input for another:
\begin{lstlisting}
echo "The line count is wc -l foo"
\end{lstlisting}
where \lstinline{foo} is the name of an existing file.

Use backquotes or command macro:
\begin{lstlisting}
echo "The line count is `wc -l foo`"
echo "There are $( wc -l foo ) lines"
\end{lstlisting}
\end{numberedframe}

\Level 0 {Permissions}

\begin{numberedframe}{Basic permissions}
  \label{sl-lnx:rwx}
  \begin{itemize}
  \item Three degrees of access: user/group/other
  \item three types of access: read/write/execute
  \end{itemize}
  \[ \begin{array}{ccc}
    \hline user&group&other\\ \hline rwx&rwx&rwx \\ \hline
  \end{array}
  \]
  Example: \lstinline{rw-r-----}:\\
  owner read-write, group read, world nothing
\end{numberedframe}

\begin{numberedframe}{Permission setting}
  \label{sl-lnx:755}
  \begin{itemize}
  \item Add permissions \lstinline{chmod g+w myfile}
  \item recursively: \lstinline{chmod -R o-r mydirectory}
  \item Permissions are an octal number:
    \lstinline{chmod 644 myfile}
  \end{itemize}
\end{numberedframe}

\begin{numberedframe}{Share files}
  \label{sl-lnx:share}
  \begin{itemize}
  \item Make a file in your \lstinline{$WORK} file system,
    and make it visible to the world.
  \item Ask a fellow student to view it.
  \item $\Rightarrow$ also necessary to make \lstinline{$WORK} readable.\\
    (Not a good idea to make \lstinline{$HOME} readable.)
  \end{itemize}
\end{numberedframe}

\begin{numberedframe}{The \texttt{x} bit}
  \label{sl-lnx:x}
  The \lstinline{x} bit has two meanings:
  \begin{itemize}
  \item For regular files: executable.
  \item For directories: you can go into them.
  \item Make all directories viewable:\\
    \lstinline{chmod -R g+X,o+X rootdir}
  \end{itemize}
\end{numberedframe}

\Level 0 {Command execution}

\begin{numberedframe}{Command execution}
  \label{sl-lnx:execute}
  \begin{itemize}
  \item Some shell commands are built-in, however most are programs.
  \item \lstinline{which ls}
  \item Exercise: what can you find out about the \lstinline{ls} program?
  \item Programs can be called directly: \lstinline{/bin/ls}
    or found along the search path \lstinline{$PATH}:\\
    \lstinline{echo $PATH}
  \end{itemize}
\end{numberedframe}

\begin{numberedframe}{The PATH variable}
  \begin{itemize}
  \item The \lstinline{PATH} variable is set by the system
  \item You can set it in the \lstinline{.bashrc} file
  \item The TACC module system sets it.
  \item Temporary changes:
\begin{lstlisting}
export PATH=/my/bin/dir:${PATH}
\end{lstlisting}
\item Changes to \lstinline{.bashrc} take effect next time you log in\\
  or \lstinline{source .bashrc} for immediate results.
  \end{itemize}
\end{numberedframe}

\begin{numberedframe}{Your \texttt{bin} directory}
  Common practice:
  \begin{itemize}
  \item Have a directory \lstinline+${HOME}/bin+
  \item Put your own scripts there
  \item Set
\begin{lstlisting}
export PATH=${HOME}/bin:${PATH}
\end{lstlisting}
        in your \lstinline{.bashrc}
  \end{itemize}
\end{numberedframe}

\begin{numberedframe}{Things that look like commands}
  \label{sl-lnx:alias}
  \begin{itemize}
  \item Use \lstinline{alias} to give a new name to a command:\\
    \lstinline{alias ls='ls -F'}\\
    \lstinline{alias rm='rm -i'}
  \item There is a shell level \lstinline{function} mechanism,
    not explained here.
  \end{itemize}
\end{numberedframe}

\begin{numberedframe}{Processes}
  \label{sl-lnx:ps}
\begin{tabular}{ll}
  \toprule
  \lstinline{ps}&list (all) processes\\
  \lstinline{kill}&kill a process\\
  \verb+CTRL-c+&kill the foreground job\\
  \verb+CTRL-z+&suspect the foreground job\\
  \lstinline{jobs}&give the status of all jobs\\
  \lstinline{fg}&bring the last suspended job to the foreground\\
  \verb+fg %3+&bring a specific job to the foreground\\
  \lstinline{bg}&run the last suspended job in the background\\
  \bottomrule
\end{tabular}

Exercise: how many programs do you have running?
\end{numberedframe}

\Level 0 {Shell programming}

\begin{numberedframe}{Variables}
  \label{sl-lnx:var}
  \begin{itemize}
  \item \lstinline{PATH} is a variable, built-in to the shell
    \item you can make your own variables:
\begin{lstlisting}
a=5
echo $a
\end{lstlisting}
No spaces around the equals!
  \end{itemize}
Exercise: what happens when you try to add two variables together?
\begin{lstlisting}
a=3
b=5
\end{lstlisting}
\end{numberedframe}

\begin{numberedframe}{Variable manipulation}
  \begin{itemize}
  \item Often you want to strip prefixes or suffixes from a variable:\\
    \lstinline{program.c} $\Rightarrow$ \lstinline{program}\\
    \lstinline{/usr/bin/program} $\Rightarrow$ \lstinline{program}
  \item Parameter expansion:
\begin{lstlisting}
a=program.c
echo ${a%%.c}
a=/foo/bar/program.c
eecho ${a##*/}
\end{lstlisting}
  \end{itemize}
\end{numberedframe}

\begin{numberedframe}{Conditionals}
  \label{sl-lnx:if}
\begin{itemize}
  \item Mostly text-based tests:
\begin{lstlisting}
if [ $a = "foo" ] ; then
  echo "that was foo"
else
  echo "that was $a"
fi
\end{lstlisting}
\item Single line:
{  \footnotesize
\begin{lstlisting}
if [ $a = "foo" ] ; then echo "foo" ; else echo "something" ; fi
\end{lstlisting}
}
Note the semicolons!\\
also spaces around square brackets.
  \end{itemize}
\end{numberedframe}

\begin{numberedframe}{Other conditionals}
  \begin{itemize}
\item Numerical tests:/
\begin{lstlisting}
if [ $a -gt 2 ] ....
\end{lstlisting}
\item Test for existence of file and directory:
{ \footnotesize
\begin{lstlisting}
if [ -f $HOME ] ; then echo "exists" ; else echo "no such" ; fi
if [ -d $HOME ] ; then echo "directory!" ; else echo "file" ; fi
\end{lstlisting}
}
  \end{itemize}
\end{numberedframe}

\begin{numberedframe}{Looping}
  \label{sl-lnx:for}
  \begin{itemize}
  \item Loop: for item in list\\
    the item is available as macro
\begin{lstlisting}
for letter in a b c ; do echo $letter ; done
\end{lstlisting}
\item Loop over files:
\begin{lstlisting}
for file in * ; do echo $file ; done
\end{lstlisting}
  \end{itemize}
  Exercises:
  \begin{enumerate}
  \item for each file, print its name and how many lines there are in it.
  \item loop through your files, print which ones are directories.
  \item for each C program, remove the object file.
  \end{enumerate}
\end{numberedframe}

\Level 0 {Scripting}

\begin{numberedframe}{Sourcing}
  \begin{itemize}
  \item Make a file with Unix commands;
  \item `source' it:
\begin{lstlisting}
source my_commands.txt      
\end{lstlisting}
  \end{itemize}
  As if you had typed the commands yourself.

  This is not a script: it affects your shell.
  Example: put \lstinline{cd} in your commands file.
\end{numberedframe}

\begin{numberedframe}{Script execution}
  \label{sl-lnx:script}
  \begin{itemize}
  \item Create a script \lstinline{script.sh}:
\begin{lstlisting}
#!/bin/bash
echo foo
\end{lstlisting}
\item Can you execute this? Does the error suggest a remedy?
\item What is the remaining problem?
  \end{itemize}
\end{numberedframe}

\begin{numberedframe}{Arguments}

  \begin{itemize}
  \item You want to call \lstinline{./script.sh myfile}
    \item Parameters are \lstinline{$1} et cetera:
\begin{lstlisting}
#!/bin/bash
echo "$1 is a file"
\end{lstlisting}
\item How many arguments: \lstinline{$#}
  \end{itemize}
\end{numberedframe}

   \end{itemize}
\end{exerciseframe}

\begin{exerciseframe}{Plagiarism detector}
  \label{sl-lnx:plag}
  Write a `plagiarism detector'.
  \begin{itemize}
  \item Write a script that accepts two argument: one text file and one directory\\
    \lstinline{./yourscript.sh myfile targetdir}\\
    (the \lstinline{.sh} extension is required for this exercise)
  \item Your script should compare the text file to the contents of the directory:
    \begin{itemize}
    \item If the file is different from anything in the directory, it
      should be copied into the directory;
      the script should not produce any output in this case.
    \item If the file is the same as a file in the directory,
      the script should complain.
    \item The test whether files are `the same' should be made with the \lstinline{diff}
      command. Explore options that allow \lstinline{diff} to ignore differences that
      are only in whitespace.
    \end{itemize}
  \end{itemize}
\end{exerciseframe}
